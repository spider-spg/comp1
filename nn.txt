Script for 2 nodes : 
# 1. Create a simulator object
set ns [new Simulator]
# Define different colors for data flows
$ns color 1 Blue
# 2. Open the nam trace file
set nf [open out.nam w]
$ns namtrace-all $nf
# 7. Define a 'finish' procedure
proc finish {} {
    global ns nf
    $ns flush-trace
    # Close the trace file
       close $nf
    # Execute nam on the trace file
    exec nam out.nam &
    exit 0
}
# 3. Create n numbers of nodes (we create 2)
set n0 [$ns node]
set n1 [$ns node]
# 4. Set the link between each of the nodes
# Create a 1Mb/s, 10ms delay, DropTail queue link between n0 and n1
$ns duplex-link $n0 $n1 1Mb 10ms DropTail
# 5. Create agent and attach it
# Create a UDP agent (source) and attach it to node n0
set udp0 [new Agent/UDP]
$udp0 set class_ 1
$ns attach-agent $n0 $udp0
# Create a Null agent (a traffic sink) and attach it to node n1
set null0 [new Agent/Null]
$ns attach-agent $n1 $null0
# Connect the traffic source to the traffic sink
$ns connect $udp0 $null0
# 6. Trace the protocols using CBR
# Create a CBR traffic source and attach it to udp0
set cbr0 [new Application/Traffic/CBR]
$cbr0 set packetSize_ 500
$cbr0 set interval_ 0.005
$cbr0 attach-agent $udp0
# 8. Schedule events for simulation
$ns at 0.5 "$cbr0 start"
$ns at 4.5 "$cbr0 stop"
# Call the finish procedure after 5 seconds
$ns at 5.0 "finish"
# 9. Run the simulation
$ns run
